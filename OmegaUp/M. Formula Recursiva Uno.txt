import java.util.Scanner;

public class Main {

    public static long f(int n) {
        if (n <= 5) {
            return 1; 
        } else {
            return f(n - 2) * 5; 
        }
    }

    public static void main(String[] args) {
        Scanner sc = new Scanner(System.in);

        int n = sc.nextInt();

        long resultado = f(n);

        System.out.println(resultado);
    }
}
